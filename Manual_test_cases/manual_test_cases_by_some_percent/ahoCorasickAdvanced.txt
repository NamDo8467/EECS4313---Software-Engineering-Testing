@Test
  public void testMatchSearchAll(){

    AhoCorasickAdvanced ac = new AhoCorasickAdvanced();
    ac.addPattern("they", 1);
    ac.addPattern("them", 2);
    ac.addPattern("we", 3);
    ac.addPattern("us", 4);
    ac.addPattern("they", 5);
    ac.build();
    
    List<Match> matches = ac.searchAll("they");
    assertEquals(2, matches.size()); // Should find two "they"s

    for (int i = 0; i < matches.size(); i++) {
      Match match = matches.get(i);
      assertEquals(0, match.getStart()); // should start at 0
      assertEquals(3, match.getEnd()); // should end at 3
      assertEquals(4, match.length()); // should have a length of 4
      // System.out.println("Start: "+ match.getStart() + " End: " + match.getEnd() + " Length: " + match.length());
    }
  }
